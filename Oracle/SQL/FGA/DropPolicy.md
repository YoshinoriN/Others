#### 作成した監査を削除する
```sql
BEGIN
	FOR TABLES_CURSOR IN (SELECT UO.OBJECT_NAME
				FROM USER_OBJECTS UO
					INNER JOIN DBA_AUDIT_POLICIES AUD_PO
						ON UO.OBJECT_NAME = AUD_PO.OBJECT_NAME
				WHERE	UO.OBJECT_TYPE = 'TABLE'
				)
	LOOP
		--監査を削除する
		--自スキーマで実行する場合[OBJECT_SCHEMA]は省略可
		BEGIN
			DBMS_FGA.DROP_POLICY(
			OBJECT_SCHEMA => '*****',
			OBJECT_NAME => TABLES_CURSOR.OBJECT_NAME,
			POLICY_NAME => TABLES_CURSOR.OBJECT_NAME
			);
		END;
	END LOOP;
END;
```
